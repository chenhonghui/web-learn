var index =0;
function calcuRoute(basePots,aimPot='d') {
  pots = JSON.parse(JSON.stringify(basePots));
  var s = {
    a: { val: 0, router: "" }
  };
  var min = { val: 1000, keyVal: null };
  var initPot = { newVal: "a", router: null };
  var routers = [];
  do {
    index++;
    const { newVal, router } = initPot;
    console.log("-----newVal:", newVal);

    for (key in pots) {
      if (key.includes(newVal)) {
        const curPot = key.replace(newVal, "");

        // a,0,

        //  sC = sC || 1000;
        const oldValue = s[curPot] ? s[curPot].val : 1000; // 当前节点已有的值
        // ac   , sB + Bc   || 0 > new sC
        const newValue = (s[newVal] ? s[newVal].val : 0) + pots[key]; // 当前节点 新算的值

        // isUpdate ,新路线值小于，老路线值；
        const isUpdate = newValue < oldValue;

        // a

        const lastRoute =
          s[newVal] && s[newVal].router ? s[newVal].router + "-" : ""; // 上次遍历到当前节点的路径；

        //  sC= aC || bC
        const oldRoute = s[curPot] ? s[curPot].router : key;

        // key=bC, ''
        const newRoute = isUpdate && key != oldRoute ? key : "";
        let upadateRoute = lastRoute + oldRoute + newRoute;
        if(key != oldRoute){
          upadateRoute = upadateRoute.replace(oldRoute,'')
        }
        
        s[curPot] = {
          val: isUpdate ? newValue : oldValue,
          router: isUpdate
            ?  upadateRoute
            : oldRoute + newRoute
        };
        min = pots[key] < min.val ? { val: pots[key], keyVal: key } : min;
        delete pots[key];
      }
    }
    // 每遍历一个节点的最小值 min
    initPot = {
      newVal: min.keyVal.replace(newVal, ""), //下次遍历所需节点；
      router: router + "-" + min.keyVal
    };
  } while (!s[aimPot]);
  // re-calculate
  do {
    for (key in pots) {
      const curPot = key.replace(aimPot, "");
      if (key.includes(aimPot) && s[curPot]) {
        const sd = pots[key] + s[curPot].val;
        if (sd < s[aimPot].val) {
          console.log("-----upadate:", s);
          s[aimPot] = {
            val: sd,
            router: s[curPot].router + "-" + key
          };
        }
      }
      delete pots[key];
    }
  } while (!!Object.keys(pots).length);
  console.log("-----end:", s[aimPot]);
}
